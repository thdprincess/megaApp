<!-- 서비스 워커 등록 -->
  <script>
    if ('serviceWorker' in navigator) {
      navigator.serviceWorker
        .register('./service_worker.js')
        .then(function () {
          console.log('서비스 워커가 등록됨!');
        })
    }
  </script>

// 브라우저가 서비스 워커를 지원하는 경우, ./service_worker.js 파일을 서비스 워커로 등록하고, 성공적으로 등록되면 콘솔에 '서비스 워커가 등록됨!'이라는 메시지를 출력하는 기능 수행
